# Phase 54: PWA Foundation & App Shell - Research

**Researched:** 2026-05-28
**Domain:** Progressive Web App — Service Worker caching, App Shell pattern, Next.js static assets
**Confidence:** HIGH

## Summary

The existing PWA infrastructure covers ~80% of requirements. The manifest, service worker registration, install prompt, and offline page are all in place and functional. The critical gap is **app shell caching** — the service worker's `SHELL_ASSETS` only pre-caches `/` and `/offline`, meaning repeat visits still require network for layout JS chunks, CSS, and fonts.

The primary challenge is that Next.js uses content-hashed filenames for chunks (e.g., `_next/static/chunks/0-19kw-pvuxnc.js`) that change on every build. A static list in `sw.js` won't work. The solution is a **build-time injection script** that reads the Next.js build manifest and writes the critical asset paths into the service worker, combined with **runtime caching** for Google Fonts and dynamic Next.js assets.

**Primary recommendation:** Add a post-build script that reads `.next/static/{buildId}/_buildManifest.js` to extract critical asset paths, injects them into `sw.js` SHELL_ASSETS array, and adds runtime caching rules for fonts and `_next/static/` resources using cache-first strategy.

### Pattern 1: Build-Time Asset Injection
**What:** Post-build script reads Next.js build manifest and injects paths into sw.js
**When to use:** When pre-caching Next.js hashed assets without a framework like Workbox
**Example:**
```javascript
// scripts/inject-sw-assets.mjs
// [ASSUMED] — pattern derived from Next.js build output structure observed in codebase
import { readFileSync, writeFileSync, readdirSync } from 'fs';
import { join } from 'path';

const NEXT_DIR = join(process.cwd(), '.next');
const SW_PATH = join(process.cwd(), 'public', 'sw.js');

// Find build ID
const buildId = readFileSync(join(NEXT_DIR, 'BUILD_ID'), 'utf8').trim();

// Read build manifest to find critical chunks
const manifestPath = join(NEXT_DIR, 'static', buildId, '_buildManifest.js');
const manifestContent = readFileSync(manifestPath, 'utf8');

// Extract JS/CSS paths from manifest (pages "/" entry = app shell)
const assetPaths = [];

// Always include the build manifest itself and ssg manifest
assetPaths.push(`/_next/static/${buildId}/_buildManifest.js`);
assetPaths.push(`/_next/static/${buildId}/_ssgManifest.js`);

// Parse page chunks for "/" route from _buildManifest.js
// The manifest self-assigns to self.__BUILD_MANIFEST with page->chunks mapping
const chunkMatch = manifestContent.match(/"\/":\s*\[([^\]]+)\]/);
if (chunkMatch) {
  const chunks = chunkMatch[1].match(/"([^"]+)"/g)?.map(s => s.replace(/"/g, ''));
  if (chunks) {
    assetPaths.push(...chunks.map(c => `/_next/${c}`));
  }
}

// Also grab shared/common chunks (framework, main, webpack runtime)
const chunksDir = join(NEXT_DIR, 'static', 'chunks');
const frameworkChunks = readdirSync(chunksDir)
  .filter(f => f.startsWith('framework-') || f.startsWith('main-') || f.match(/^webpack-/))
  .map(f => `/_next/static/chunks/${f}`);
assetPaths.push(...frameworkChunks);

// Inject into sw.js
const swContent = readFileSync(SW_PATH, 'utf8');
const shellAssets = JSON.stringify([
  '/',
  '/offline',
  ...new Set(assetPaths),
], null, 2);

const updatedSw = swContent.replace(
  /const SHELL_ASSETS = \[[\s\S]*?\];/,
  `const SHELL_ASSETS = ${shellAssets};`
);

// Bump cache version
const newVersion = `peerlight-shell-v${Date.now()}`;
const finalSw = updatedSw.replace(
  /const CACHE_NAME = '[^']+';/,
  `const CACHE_NAME = '${newVersion}';`
);

writeFileSync(SW_PATH, finalSw);
console.log(`SW updated: ${assetPaths.length} assets, cache: ${newVersion}`);
```

### Pattern 2: Runtime Caching Strategy (Enhanced sw.js)
**What:** Differentiated caching strategies per URL pattern
**When to use:** When different resource types need different freshness guarantees
**Example:**
```javascript
// Enhanced fetch handler for sw.js
// [ASSUMED] — standard Service Worker patterns from MDN Web Docs
self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);

  // Never cache API calls (privacy requirement)
  if (url.pathname.startsWith('/api/')) return;

  // Google Fonts: stale-while-revalidate
  if (url.hostname === 'fonts.googleapis.com' || url.hostname === 'fonts.gstatic.com') {
    event.respondWith(
      caches.open('peerlight-fonts-v1').then(async (cache) => {
        const cached = await cache.match(request);
        const fetchPromise = fetch(request).then((response) => {
          if (response.ok) cache.put(request, response.clone());
          return response;
        }).catch(() => cached);
        return cached || fetchPromise;
      })
    );
    return;
  }

  // Next.js static assets: cache-first (hashed = immutable)
  if (url.pathname.startsWith('/_next/static/')) {
    event.respondWith(
      caches.match(request).then((cached) => {
        if (cached) return cached;
        return fetch(request).then((response) => {
          if (response.ok) {
            const clone = response.clone();
            caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
          }
          return response;
        });
      })
    );
    return;
  }

  // Navigation requests: network-first, fallback to offline
  if (request.mode === 'navigate') {
    event.respondWith(
      fetch(request).catch(() => caches.match('/offline'))
    );
    return;
  }

  // All other static: cache-first
  event.respondWith(
    caches.match(request).then((cached) => {
      return cached || fetch(request).catch(() => undefined);
    })
  );
});
```

### Anti-Patterns to Avoid
- **Caching API responses:** Privacy violation — student psychological data must never be stored in browser cache. The `/api/` exclusion must remain absolute. [CITED: REQUIREMENTS.md constraints]
- **Static SHELL_ASSETS with hashed names:** Will break on every deployment since Next.js chunk hashes change per build. Must use build-time injection.
- **Caching HTML pages aggressively:** Next.js pages include session-specific data. Only navigation fallback to `/offline` is safe.
- **Using importScripts in SW for large libraries:** Adds latency to SW boot. Keep sw.js self-contained and small.

## Common Pitfalls

### Pitfall 1: Stale Service Worker Blocking Updates
**What goes wrong:** Users stuck on old cached version because SW doesn't update
**Why it happens:** Browser only checks for SW updates on navigation; cached responses prevent seeing new content
**How to avoid:** `skipWaiting()` + `clients.claim()` (already implemented); unique CACHE_NAME per build ensures old caches are evicted on activate
**Warning signs:** Users reporting they don't see new features after deploy

### Pitfall 2: Over-Caching Breaking Auth Flow
**What goes wrong:** Cached pages served to logged-out users, or login page cached with session data
**Why it happens:** Cache-first on HTML navigation responses
**How to avoid:** Network-first for all navigation requests (HTML pages); only cache-first for static assets. Current pattern is correct — maintain it.
**Warning signs:** Users seeing authenticated content after logout

### Pitfall 3: Google Fonts CORS Issues in Cache
**What goes wrong:** Fonts fetched with `no-cors` can't be read from cache properly
**Why it happens:** Cross-origin font files have opaque responses that consume quota
**How to avoid:** Google Fonts sets proper CORS headers for font files (fonts.gstatic.com). The CSS from fonts.googleapis.com is also CORS-enabled. Standard fetch works correctly.
**Warning signs:** Fonts not rendering on cached pages; storage quota warnings

### Pitfall 4: Build Script Runs Before Next.js Output Exists
**What goes wrong:** Post-build script fails because .next directory doesn't exist yet
**Why it happens:** Script hooked into wrong lifecycle (pre-build instead of post-build)
**How to avoid:** Chain as `"build": "next build && node scripts/inject-sw-assets.mjs"` in package.json
**Warning signs:** Empty SHELL_ASSETS array in deployed sw.js

### Pitfall 5: SW Pre-cache List Too Large
**What goes wrong:** Initial page load takes too long downloading all chunks; install event times out
**Why it happens:** Trying to pre-cache ALL page chunks instead of just the shell (framework, layout, root page)
**How to avoid:** Only pre-cache framework chunks + root route chunks. Let other pages cache at runtime via cache-first fetch handler.
**Warning signs:** SW install event taking >3s; competing with user's initial page load for bandwidth

## Code Examples

### Enhanced sw.js Template (Final Form)
```javascript
// Source: Codebase analysis + MDN Service Worker patterns [ASSUMED]
const CACHE_NAME = 'peerlight-shell-v1'; // Replaced at build time
const FONTS_CACHE = 'peerlight-fonts-v1';
const SHELL_ASSETS = [
  '/',
  '/offline',
  // Additional assets injected at build time by scripts/inject-sw-assets.mjs
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(SHELL_ASSETS))
  );
  self.skipWaiting();
});

self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(
        keys
          .filter((k) => k !== CACHE_NAME && k !== FONTS_CACHE)
          .map((k) => caches.delete(k))
      )
    )
  );
  self.clients.claim();
});

self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);

  // PRIVACY: Never cache API calls
  if (url.pathname.startsWith('/api/')) return;

  // Google Fonts: stale-while-revalidate
  if (url.hostname === 'fonts.googleapis.com' || url.hostname === 'fonts.gstatic.com') {
    event.respondWith(
      caches.open(FONTS_CACHE).then(async (cache) => {
        const cached = await cache.match(request);
        const networkPromise = fetch(request)
          .then((res) => { if (res.ok) cache.put(request, res.clone()); return res; })
          .catch(() => cached);
        return cached || networkPromise;
      })
    );
    return;
  }

  // Next.js hashed static assets: cache-first (immutable by hash)
  if (url.pathname.startsWith('/_next/static/')) {
    event.respondWith(
      caches.match(request).then((cached) => {
        if (cached) return cached;
        return fetch(request).then((res) => {
          if (res.ok) {
            const clone = res.clone();
            caches.open(CACHE_NAME).then((c) => c.put(request, clone));
          }
          return res;
        });
      })
    );
    return;
  }

  // Navigations: network-first with offline fallback
  if (request.mode === 'navigate') {
    event.respondWith(
      fetch(request).catch(() => caches.match('/offline'))
    );
    return;
  }

  // Other static assets (icons, images): cache-first
  event.respondWith(
    caches.match(request).then((cached) =>
      cached || fetch(request).catch(() => undefined)
    )
  );
});
```
